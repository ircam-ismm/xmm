162  template <typename T>
163  void json2array(JSONNode root, T* a, int n)
164  {
165  if (root.type() != JSON_ARRAY)
166  throw JSONException("Wrong type: was expecting 'JSON_ARRAY'", root.name());
167  unsigned int i = 0;
168  for (JSONNode::const_iterator array_it = root.begin(); array_it != root.end(); ++array_it)
169  {
170  if (i >= n)
171  throw JSONException("JSON 2 Array: Index out of bounds");
172  if (array_it->type() != JSON_NUMBER)
173  throw JSONException("JSON 2 Vector: Wrong type");
174  a[i++] = array_it->as_int();
175  }
176  }
